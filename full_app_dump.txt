67:const getRandomColor = () => {
68:  const h = Math.floor(Math.random() * 360); // 0 to 359 (entire color wheel)
69:  const s = Math.floor(Math.random() * 60) + 30; // 30% to 90% saturation
70:  const l = Math.floor(Math.random() * 20) + 15; // 15% to 35% lightness for white text
71:  return `hsl(${h}, ${s}%, ${l}%)`;
72:};

74:const darkenHsl = (hslStr: string, percentage: number) => {
75:  if (!hslStr.startsWith("hsl")) return hslStr;
76:  const match = hslStr.match(/hsl\((\d+),\s*(\d+)%,\s*(\d+)%\)/);
77:  if (!match) return hslStr;
78:  const h = parseInt(match[1]);
79:  const s = parseInt(match[2]);
80:  const l = Math.max(0, parseInt(match[3]) - percentage);
81:  return `hsl(${h}, ${s}%, ${l}%)`;
82:};
